SET(LIB_NAME SpatialViz)

    SpatialViz.h
)

    STATIC
    ${PLUGIN_HEADERS}
    SpatialViz.cpp
)

TARGET_LINK_LIBRARIES(${LIB_NAME} ${PHYSX_LIBRARIES})

ADD_CALVR_LIBRARIES(${LIB_NAME})

INSTALL(TARGETS ${LIB_NAME} DESTINATION lib/plugins)
